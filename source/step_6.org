#+begin_src python   
import turtle

wn = turtle.Screen()
wn.bgcolor("white") 
wn.title("Mystery Command")

tess = turtle.Turtle()
tess.penup()

for n in range(1,11):
    tess.color('red')
    tess.stamp()
    tess.forward(50)  # Draw a square in the current colour.        
    tess.left(360/10)
    
wn.mainloop()
#+end_src
